// FITUR GRUP CHAT function createGroup() { const gName = prompt("Nama Grup Baru:"); if (!gName) return; const gid = "g" + Date.now(); db.ref('groups/' + gid).set({ id: gid, name: gName, admin: currentUser.uid, members: [currentUser.uid], lastMsg: "Grup dibuat", time: Date.now() }).then(() => alert("Grup " + gName + " Aktif!")); } function listenGroups() { db.ref('groups').on('value', snap => { const box = document.getElementById('list-chats'); // Gabung di daftar chat snap.forEach(child => { const g = child.val(); if (g.members.includes(currentUser.uid)) { box.innerHTML += `
${g.name} (Grup)
${g.lastMsg}
`; } }); }); } function openGroup(gid, name) { activeChatId = gid; document.getElementById('chat-name').innerText = name; document.getElementById('chat-area').classList.add('active'); loadMsgs(); } // UPDATE STATUS ONLINE OTOMATIS function updateStatus() { const statusRef = db.ref('users/' + currentUser.uid); statusRef.update({ status: 'online' }); statusRef.onDisconnect().update({ status: 'offline', lastSeen: Date.now() }); } // Menampilkan daftar permintaan teman secara visual function renderRequestList() { const listArea = document.getElementById('list-reqs'); if(!listArea) return; db.ref('friend_requests').orderByChild('receiver').equalTo(currentUser.uid).on('value', snap => { listArea.innerHTML = ""; let count = 0; snap.forEach(child => { const req = child.val(); if(req.status === 'pending') { count++; db.ref('users/' + req.sender).once('value', uSnap => { const u = uSnap.val(); listArea.innerHTML += `
${u.username}
"${u.bio}"
`; }); } }); const badge = document.getElementById('req-badge'); if(badge) { badge.innerText = count; badge.classList.toggle('hidden', count === 0); } }); } // Update Profil & Bio (Biar Bio lo ga cuma pajangan) async function updateProfile() { const newBio = prompt("Masukkan Bio Baru:", currentUser.bio); if(newBio !== null) { await db.ref('users/' + currentUser.uid).update({ bio: newBio }); currentUser.bio = newBio; localStorage.setItem('zchat_session', JSON.stringify(currentUser)); alert("Bio diperbarui!"); } }
